15  /**
16   * KomennonKuuntelija kuuntelee pelaajan antamia komentoja näppäimistöltä.
17   */
18  public class KomennonKuuntelija implements KeyListener {
19  
20      private KomennonKasittelija kasittelija;
21      private KomennonValidoija validoija;
22      private Pelaaja pelaaja;
23      private Kyla kyla;
24  
25      public KomennonKuuntelija(Kyla kyla, Pelaaja pelaaja, TarinaPanel tarinaPanel, PelaajaTietoPanel pelaajaTietoPanel, Metsa metsa, Areena areena, Luola luola, Parantaja parantaja) {
26          this.pelaaja = pelaaja;
27          this.kasittelija = new KomennonKasittelija(kyla, pelaaja, tarinaPanel, pelaajaTietoPanel, metsa, areena, luola, parantaja);
28          this.validoija = new KomennonValidoija(pelaaja);
29          this.kyla = kyla;
30      }
31  
32      @Override
33      public void keyTyped(KeyEvent e) {
34      }
35  
36      @Override
37      public void keyPressed(KeyEvent e) {
38          if (validoija.onkoKomento(pelaaja.getPaikka(), e.getKeyCode())) {
39              kasittelija.kasitteleKomento(pelaaja.getPaikka(), e.getKeyCode());
40          } else if (pelaaja.getPaikka() == Paikka.KYLA && e.getKeyCode() == KeyEvent.VK_ESCAPE) {
41              kyla.pelistaValikkoon();
42          }
43      }
44  
45      @Override
46      public void keyReleased(KeyEvent e) {
47      }
48  
49  }
