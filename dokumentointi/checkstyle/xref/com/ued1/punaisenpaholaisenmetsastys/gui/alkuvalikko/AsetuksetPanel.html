3   import com.ued1.punaisenpaholaisenmetsastys.gui.Kyla;
4   import com.ued1.punaisenpaholaisenmetsastys.gui.alkuvalikko.kuuntelijat.AsetustenVaihtaja;
5   import com.ued1.punaisenpaholaisenmetsastys.peli.hahmot.Pelaaja;
6   import com.ued1.punaisenpaholaisenmetsastys.peli.Vaikeus;
7   import java.awt.Color;
8   import static java.awt.Component.CENTER_ALIGNMENT;
9   import java.awt.Font;
10  import javax.swing.BoxLayout;
11  import javax.swing.ButtonGroup;
12  import javax.swing.JButton;
13  import javax.swing.JLabel;
14  import javax.swing.JPanel;
15  import javax.swing.JRadioButton;
16  
17  /**
18   * Asetusvalikko-paneeli. Asetusten vaihtaminen pelin ollessa käynnissä.
19   */
20  public class AsetuksetPanel extends JPanel {
21  
22      private Kyla kyla;
23      private Pelaaja pelaaja;
24      private JRadioButton normaali;
25      private JRadioButton helppo;
26      private ButtonGroup valintaNapit;
27      private JButton takaisin;
28      private JLabel valitseVaikeus;
29      private AsetustenVaihtaja asetustenVaihtaja;
30  
31      public AsetuksetPanel(Kyla kyla, Pelaaja pelaaja) {
32          this.kyla = kyla;
33          this.pelaaja = pelaaja;
34          luoNapit();
35          this.asetustenVaihtaja = new AsetustenVaihtaja(kyla, pelaaja, takaisin, normaali, helppo);
36          luoSisalto();
37      }
38  
39      private void luoNapit() {
40          if (pelaaja.getVaikeus() == Vaikeus.HELPPO) {
41              normaali = new JRadioButton("Normaali", false);
42              helppo = new JRadioButton("Helppo", true);
43          } else {
44              normaali = new JRadioButton("Normaali", true);
45              helppo = new JRadioButton("Helppo", false);
46          }
47          takaisin = new JButton("Takaisin");
48      }
49  
50      private void luoSisalto() {
51          valintaNapit = new ButtonGroup();
52          valitseVaikeus = new JLabel("Vaikeusasteen vaihtaminen");
53  
54          komponenttienAsetukset();
55  
56          valintaNapit.add(normaali);
57          valintaNapit.add(helppo);
58          lisaaKuuntelijat();
59          lisaaKomponentit();
60      }
61  
62      private void komponenttienAsetukset() {
63          setLayout(new BoxLayout(this, BoxLayout.Y_AXIS));
64          setBackground(Color.BLACK);
65  
66          normaali.setBackground(Color.BLACK);
67          normaali.setForeground(Color.WHITE);
68          helppo.setBackground(Color.BLACK);
69          helppo.setForeground(Color.WHITE);
70          normaali.setAlignmentX(CENTER_ALIGNMENT);
71          helppo.setAlignmentX(CENTER_ALIGNMENT);
72          takaisin.setAlignmentX(CENTER_ALIGNMENT);
73  
74          valitseVaikeus.setFont(new Font("Dialog", Font.BOLD, 15));
75          valitseVaikeus.setForeground(Color.WHITE);
76          valitseVaikeus.setAlignmentX(CENTER_ALIGNMENT);
77  
78      }
79  
80      private void lisaaKuuntelijat() {
81          takaisin.addActionListener(asetustenVaihtaja);
82          normaali.addActionListener(asetustenVaihtaja);
83          helppo.addActionListener(asetustenVaihtaja);
84      }
85  
86      private void lisaaKomponentit() {
87          add(new JLabel(""));
88          add(valitseVaikeus);
89          add(new JLabel(""));
90          add(normaali);
91          add(helppo);
92          add(new JLabel(""));
93          add(takaisin);
94      }
95  
96  }
