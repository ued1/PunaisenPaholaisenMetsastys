5   import com.ued1.punaisenpaholaisenmetsastys.tyokalut.KuvanAsettaja;
6   import com.ued1.punaisenpaholaisenmetsastys.peli.hahmot.Pelaaja;
7   import java.awt.Color;
8   import static java.awt.Component.CENTER_ALIGNMENT;
9   import java.awt.Dimension;
10  import javax.swing.BoxLayout;
11  import javax.swing.JButton;
12  import javax.swing.JLabel;
13  import javax.swing.JPanel;
14  import javax.swing.JTextArea;
15  
16  /**
17   * LoppuPanel näytetään pelin lopussa kun pelaaja on voittanut Punaisen
18   * Paholaisen.
19   */
20  public class LoppuPanel extends JPanel {
21  
22      private JButton lopeta;
23      private JButton alkuvalikkoon;
24      private Pelaaja pelaaja;
25      private JLabel loppukuva;
26      private JTextArea lopputeksti;
27      private LoppuPaneelinKuuntelija loppuPaneelinKuuntelija;
28  
29      public LoppuPanel(Kyla kyla, Pelaaja pelaaja) {
30          this.pelaaja = pelaaja;
31          this.lopputeksti = new JTextArea(luoLopputeksti());
32          this.loppukuva = new JLabel();
33          luoNapit();
34          loppuPaneelinKuuntelija = new LoppuPaneelinKuuntelija(kyla, alkuvalikkoon, lopeta);
35          alusta();
36      }
37  
38      private void alusta() {
39          setLayout(new BoxLayout(this, BoxLayout.Y_AXIS));
40          setBackground(Color.BLACK);
41          new KuvanAsettaja().asetaKuva(loppukuva, "Loppukuva");
42  
43          lopputeksti.setEditable(false);
44          lopputeksti.setPreferredSize(new Dimension(300, 75));
45          lopputeksti.setMaximumSize(new Dimension(300, 75));
46          lopputeksti.setMinimumSize(new Dimension(300, 75));
47          lopputeksti.setAlignmentX(CENTER_ALIGNMENT);
48          lopputeksti.setBackground(Color.BLACK);
49          lopputeksti.setForeground(Color.WHITE);
50          loppukuva.setBackground(Color.BLACK);
51          loppukuva.setAlignmentX(CENTER_ALIGNMENT);
52  
53          lisaaKuuntelijat();
54          lisaaKomponentit();
55      }
56  
57      private void luoNapit() {
58          lopeta = new JButton("Lopeta peli");
59          alkuvalikkoon = new JButton("Alkuvalikkoon");
60          lopeta.setAlignmentX(CENTER_ALIGNMENT);
61          alkuvalikkoon.setAlignmentX(CENTER_ALIGNMENT);
62      }
63  
64      private void lisaaKomponentit() {
65          add(loppukuva);
66          add(new JLabel(" "));
67          add(lopputeksti);
68          add(new JLabel(" "));
69          add(alkuvalikkoon);
70          add(new JLabel(" "));
71          add(lopeta);
72      }
73  
74      private void lisaaKuuntelijat() {
75          lopeta.addActionListener(loppuPaneelinKuuntelija);
76          alkuvalikkoon.addActionListener(loppuPaneelinKuuntelija);
77      }
78  
79      private String luoLopputeksti() {
80          String teksti = "\nOnneksi olkoon " + pelaaja.getNimi() + ",";
81          teksti += "\nolet selvinnyt pelin loppuun!";
82          return teksti;
83      }
84  
85  }
