16  
17  /**
18   * NimenValintaPanel on JPanel nimenvalintaa varten.
19   */
20  public class NimenValintaPanel extends JPanel {
21  
22      private Kyla kyla;
23      private JButton aloita;
24      private JButton takaisin;
25      private JRadioButton normaali;
26      private JRadioButton helppo;
27      private JTextField nimikentta;
28      private JTextArea virhekentta;
29      private JLabel valitseNimi;
30      private JLabel valitseVaikeus;
31      private ButtonGroup valintaNapit;
32      private NimenValinnanKuuntelija kuuntelija;
33      private final String ohje;
34  
35      public NimenValintaPanel(Kyla kyla) {
36          this.kyla = kyla;
37          this.ohje = luoOhje();
38          this.aloita = new JButton("Aloita peli");
39          this.takaisin = new JButton("Takaisin");
40          this.normaali = new JRadioButton("Normaali", true);
41          this.helppo = new JRadioButton("Helppo", false);
42          this.valintaNapit = new ButtonGroup();
43          this.nimikentta = new JTextField("");
44          this.virhekentta = new JTextArea(ohje);
45          this.kuuntelija = new NimenValinnanKuuntelija(this, kyla, aloita, takaisin, nimikentta, virhekentta, normaali, helppo);
46          alusta();
47      }
48  
49      private String luoOhje() {
50          String ohjeTeksti = "Nimen tulee olla pituudeltaan 2-15 merkkiä";
51          ohjeTeksti += "\npitkä ja koostua suomen kielen aakkosiin";
52          ohjeTeksti += "\nkuuluvista kirjaimista. Nimessä ei saa olla";
53          ohjeTeksti += "\nvälilyöntiä.";
54          return ohjeTeksti;
55      }
56  
57      private void alusta() {
58  
59          setLayout(new BoxLayout(this, BoxLayout.Y_AXIS));
60          setBackground(Color.BLACK);
61  
62          nimikentta.setMaximumSize(new Dimension(100, 25));
63          nimikentta.setAlignmentX(CENTER_ALIGNMENT);
64          virhekentta.setEditable(false);
65          virhekentta.setPreferredSize(new Dimension(300, 75));
66          virhekentta.setMaximumSize(new Dimension(300, 75));
67          virhekentta.setMinimumSize(new Dimension(300, 75));
68          virhekentta.setAlignmentX(CENTER_ALIGNMENT);
69  
70          valitseNimi = new JLabel("Valitse pelaajanimesi");
71          valitseNimi.setFont(new Font("Dialog", Font.BOLD, 15));
72          valitseNimi.setForeground(Color.RED);
73          valitseNimi.setAlignmentX(CENTER_ALIGNMENT);
74  
75          valitseVaikeus = new JLabel("Valitse vaikeusaste");
76          valitseVaikeus.setFont(new Font("Dialog", Font.BOLD, 15));
77          valitseVaikeus.setForeground(Color.RED);
78          valitseVaikeus.setAlignmentX(CENTER_ALIGNMENT);
79  
80          valintaNapit.add(normaali);
81          valintaNapit.add(helppo);
82          normaali.setBackground(Color.BLACK);
83          normaali.setForeground(Color.WHITE);
84          helppo.setBackground(Color.BLACK);
85          helppo.setForeground(Color.WHITE);
86  
87          asetaNormaaliTila();
88          lisaaKuuntelijat();
89          lisaaKomponentit();
90  
91      }
92  
93      private void lisaaKomponentit() {
94          add(new JLabel(" "));
95          add(new JLabel(" "));
96          add(valitseNimi);
97          add(new JLabel(" "));
98          add(nimikentta);
99          add(new JLabel(" "));
100         add(virhekentta);
101         add(new JLabel(" "));
102         add(valitseVaikeus);
103         add(new JLabel(" "));
104         add(normaali);
105         add(helppo);
106         add(new JLabel(" "));
107         add(new JLabel(" "));
108         add(aloita);
109         add(new JLabel(" "));
110         add(takaisin);
111     }
112 
113     private void lisaaKuuntelijat() {
114         nimikentta.addActionListener(kuuntelija);
115         aloita.addActionListener(kuuntelija);
116         takaisin.addActionListener(kuuntelija);
117         normaali.addActionListener(kuuntelija);
118         helppo.addActionListener(kuuntelija);
119     }
120 
121     /**
122      * Asettaa epäkelvosta pelaajanimestä johtuvat virhetilan näkyviin.
123      */
124     public void asetaVirheTila() {
125         virhekentta.setText(ohje);
126         virhekentta.setForeground(Color.WHITE);
127         virhekentta.setBackground(Color.RED);
128     }
129 
130     /**
131      * Asettaa epäkelvosta pelaajanimestä johtuneen virhetilan takaisin
132      * normaaliksi.
133      */
134     public void asetaNormaaliTila() {
135         virhekentta.setForeground(Color.WHITE);
136         virhekentta.setBackground(Color.BLACK);
137     }
138 
139 }
