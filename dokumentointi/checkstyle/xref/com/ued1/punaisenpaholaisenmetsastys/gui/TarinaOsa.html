60      private void luoKomponentit() {
61          eka = new JLabel();
62          toka = new JTextArea();
63          eka.setBackground(new Color(238, 238, 238));
64          toka.setBackground(new Color(238, 238, 238));
65          paivita();
66          eka.setFocusable(false);
67          toka.setFocusable(false);
68          add(eka);
69          add(toka);
70      }

72      /**
73       * Metodi päivittää tarinapaneelin tarinaosan.
74       */
75      public void paivita() {
76          toka.setFont(fontti);
77          if (pelaaja.getPaikka() == Paikka.KYLA) {
78              asetaKyla();
79          } else if (pelaaja.getPaikka() == Paikka.ASEPAJA) {
80              asetaAsepaja();
81          } else if (pelaaja.getPaikka() == Paikka.HAARNISKAKAUPPA) {
82              asetaHaarniskaKauppa();
83          } else if (pelaaja.getPaikka() == Paikka.METSA || pelaaja.getPaikka() == Paikka.MONSTERITAISTELUVOITTO) {
84              asetaMetsa(pelaaja.getPaikka());
85          } else if (pelaaja.getPaikka() == Paikka.ASEENOSTO) {
86              asetaAseenOsto();
87          } else if (pelaaja.getPaikka() == Paikka.ASEENMYYNTI) {
88              asetaAseenMyynti();
89          } else if (pelaaja.getPaikka() == Paikka.MONSTERITAISTELU) {
90              asetaMonsteriTaistelu();
91          } else if (pelaaja.getPaikka() == Paikka.MONSTERITAISTELUTAPPIO) {
92              asetaMonsteriTaisteluTappio();
93          } else if (pelaaja.getPaikka() == Paikka.TAISTELUAREENA) {
94              asetaTaisteluAreena();
95          } else if (pelaaja.getPaikka() == Paikka.TAISTELUAREENAEI) {
96              asetaTaisteluAreenaEi();
97          } else if (pelaaja.getPaikka() == Paikka.AREENATAISTELU) {
98              asetaAreenaTaistelu();
99          } else if (pelaaja.getPaikka() == Paikka.TAISTELUAREENATULOS) {
100             asetaTaisteluAreenaTulos();
101         } else if (pelaaja.getPaikka() == Paikka.KAPAKKA) {
102             asetaKapakka();
103         } else if (pelaaja.getPaikka() == Paikka.LUOLA) {
104             asetaLuola();
105         } else if (pelaaja.getPaikka() == Paikka.PAHOLAINEN) {
106             asetaPaholainen();
107         } else if (pelaaja.getPaikka() == Paikka.PAHOLAINENTAPPIO) {
108             asetaPaholainenTappio();
109         } else if (pelaaja.getPaikka() == Paikka.KAPAKKAOSTO) {
110             asetaKapakkaOsto();
111         } else if (pelaaja.getPaikka() == Paikka.KANNI) {
112             asetaKanni();
113         } else if (pelaaja.getPaikka() == Paikka.HAARNISKANOSTO) {
114             asetaHaarniskaOsto();
115         } else if (pelaaja.getPaikka() == Paikka.PARANTAJA) {
116             asetaParantaja();
117         } else if (pelaaja.getPaikka() == Paikka.CASINO) {
118             asetaCasino();
119         } else if (pelaaja.getPaikka() == Paikka.PEUKKUPELI) {
120             asetaPeukkupeli();
121         } else if (pelaaja.getPaikka() == Paikka.PEUKKUTULOS) {
122             asetaPeukkupeliTulos();
123         } else if (pelaaja.getPaikka() == Paikka.METSAPUPU) {
124             asetaMetsaPupu();
125         } else if (pelaaja.getPaikka() == Paikka.RUOSTEINENAVAIN) {
126             asetaRuosteinenAvain();
127         } else if (pelaaja.getPaikka() == Paikka.HEIKENNETTYPAHOLAINEN) {
128             asetaHeikennettyPaholainen();
129         } else if (pelaaja.getPaikka() == Paikka.LOPPU) {
130             asetaLopetusLyonti();
131         }
132     }

135         kuvanAsettaja.asetaKuva(eka, "Kylä");

141         kuvaus += "\nja lukea tarkemmat kuvaukset ja ohjeet";
142         kuvaus += "\neri toiminnoista.";

144             kuvaus += "\n\nSinulla ei ole yhtään rahaa.";
145             kuvaus += "\nTaistelemalla monstereita vastaan";
146             kuvaus += "\nmetsässä voit ansaita lisää rahaa.";
147         }
148         toka.setText(kuvaus);
149     }

225     private void asetaMetsa(Paikka paikka) {
226         kuvanAsettaja.asetaKuva(eka, "Metsä");
227         String kuvaus = "";
228         if (paikka == Paikka.METSA) {
229             kuvaus = "\nMetsässä voit taistella monstereita vastaan";
230             kuvaus += "\nja ansaita kultarahoja. Jos häviät monsterille";
231             kuvaus += "\nmenetät kaikki rahasi.";
232         } else if (paikka == Paikka.MONSTERITAISTELUVOITTO) {
233             kuvaus += "\nVoitit monsterin! Sait lisää rahaa";
234             kuvaus += "\nja kokemusta.";
235         }
236 
237         if (pelaaja.getPotionit() > 0) {
238             kuvaus += "\n\nVoit käyttää VointiPotioneja";
239             kuvaus += "\nparantaakseen vointiasi kun et ole";
240             kuvaus += "\ntaistelussa.";
241         } else {
242             kuvaus += "\n\nTarvitset kylän Parantajalta";
243             kuvaus += "\nVointiPotioneja, jotta voit parantaa";
244             kuvaus += "\nitseäsi metsässä taistelujen välissä.";
245         }
246         toka.setText(kuvaus);
247     }

294     private void asetaMonsteriTaistelu() {
295         String tilanne = metsa.getTaistelu().vastustaja().tiedotMerkkijonona();
296         if (metsa.getTaistelu().onkoKaynnissa()) {
297             tilanne += "\n\nOsuit monsteriin: " + metsa.getTaistelu().getEkaIsku() + " osumapistettä";
298             tilanne += "\nMonsteri";
299             if (metsa.getTaistelu().getTokaIsku() < 1) {
300                 tilanne += " ei osunut sinuun.";
301             } else {
302                 tilanne += " osui sinuun: " + metsa.getTaistelu().getTokaIsku() + " osumapistettä";
303             }
304             tilanne += "\n\nVointisi: " + pelaaja.getVointi() + "/" + pelaaja.getMaxVointi();
305         }
306         toka.setText(tilanne);
307     }

373             tokateksti += "\nPaholaisen. Löydät paholaisen metsästä.";
374         } else {
375             tokateksti += ". Haastaaksesi kilpailijoita\nja noustaksesi seuraavalle tasolle tarvitset";
376             tokateksti += "\n" + (areena.seuraavanTasonKokemus() - pelaaja.getKokemus()) + " lisää kokemusta.";
377             tokateksti += "\n\nKokemuksesi: " + pelaaja.getKokemus();
378             tokateksti += "\nTarvittava kokemus: " + areena.seuraavanTasonKokemus();

383     private void asetaTaisteluAreenaTulos() {
384         String tokateksti = "";
385         if (pelaaja.onkoElossa()) {
386             kuvanAsettaja.asetaKuva(eka, "Taisteluvoitto");
387             tokateksti += "Voitit taistelun!\n\n";
388             tokateksti += "Voit palata takaisin areenaan heti kun\n";
389             tokateksti += "koet olevasi valmis seuraavaan taisteluun.";
390         } else {
391             kuvanAsettaja.asetaKuva(eka, "Taistelutappio");
392             tokateksti += "Hävisit taistelun.\n\nVoit halutessasi yrittää heti uudestaan,";
393             tokateksti += "\nmutta paremmilla varusteilla voisi taistelu";
394             tokateksti += "\nolla helpompaa!";
395             tokateksti += "\n\nVastustajasi loukkasi itseään, joten hän";
396             tokateksti += "\non hieman heikommassa kunnossa taistelun";
397             tokateksti += "\nalkaessa uudestaan.";
398         }
399         toka.setText(tokateksti);
400         areena.asetaTaistelunTulos();
