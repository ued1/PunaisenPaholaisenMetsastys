19  /**
20   * KomentoOsa on JPanel TarinaPanel-luokan paneelissa. Luokan tehtävänä on
21   * piirtää ja päivittää komentovalikko jokaiseen tilanteeseen sopivaksi.
22   */
23  public class KomentoOsa extends JPanel {
24  
25      private JTextArea komentoValikko;
26      private Pelaaja pelaaja;
27      private Font fontti;
28      private Luola luola;
29  
30      public KomentoOsa(Pelaaja pelaaja, Luola luola) {
31          this.pelaaja = pelaaja;
32          this.luola = luola;
33          setBackground(new Color(238, 238, 238));
34          fontti = new Font("SansSerif", Font.BOLD, 12);
35          lisaaKomentoValikko();
36      }
37  
38      private void lisaaKomentoValikko() {
39          komentoValikko = new JTextArea(kylaValikko());
40          komentoValikko.setFocusable(false);
41          komentoValikko.setBackground(new Color(238, 238, 238));
42          komentoValikko.setFont(new Font("Monospaced", Font.BOLD, 12));
43          add(komentoValikko);
44      }
45  
46      private String kylaValikko() {
47          String valikko = String.format("%-18s%-12s", "[M]etsä", "[A]sepaja");
48          valikko += "\n" + String.format("%-18s%-12s", "[H]aarniskakauppa", "[P]arantaja");
49          valikko += "\n" + String.format("%-18s%-12s", "[T]aisteluareena", "[C]asino");
50          valikko += "\n" + String.format("%-18s", "[K]apakka");
51          return valikko += "\n\n[ESC] Takaisin valikkoon";
52      }
53  
54      /**
55       * Metodi päivittää komentovalikon vastaamaan pelitilannetta.
56       *
57       * @param uusiPaikka uusi paikka, jonka komennot piirretään
58       */
59      public void paivita(Paikka uusiPaikka) {
60          komentoValikko.setFont(fontti);
61          if (uusiPaikka == Paikka.KYLA) {
62              komentoValikko.setFont(new Font("Monospaced", Font.BOLD, 12));
63              komentoValikko.setText(kylaValikko());
64          } else if (uusiPaikka == Paikka.ASEPAJA) {
65              komentoValikko.setText("[O]sta ase\n[M]yy ase\n[T]akaisin");
66          } else if (uusiPaikka == Paikka.HAARNISKAKAUPPA) {
67              komentoValikko.setText("[O]sta haarniska\n[T]akaisin");
68          } else if (uusiPaikka == Paikka.METSA || uusiPaikka == Paikka.MONSTERITAISTELUVOITTO) {
69              komentoValikko.setText(lisaaMetsaKomennot());
70          } else if (uusiPaikka == Paikka.TAISTELUAREENA) {
71              komentoValikko.setText("[A]stu taisteluareenaan\n[T]akaisin kylään");
72          } else if (uusiPaikka == Paikka.ASEENOSTO) {
73              String ostettavatAseet = new Asepaja().ostokomennot(pelaaja);
74              komentoValikko.setFont(new Font("Monospaced", Font.BOLD, 12));
75              komentoValikko.setText(ostettavatAseet);
76          } else if (uusiPaikka == Paikka.ASEENMYYNTI) {
77              komentoValikko.setText(lisaaAseenMyyntiKomento());
78          } else if (uusiPaikka == Paikka.HAARNISKANOSTO) {
79              String ostettavatHaarniskat = new Haarniskakauppa().ostokomennot(pelaaja);
80              komentoValikko.setFont(new Font("Monospaced", Font.BOLD, 12));
81              komentoValikko.setText(ostettavatHaarniskat);
82          } else if (uusiPaikka == Paikka.MONSTERITAISTELU) {
83              komentoValikko.setText("[L]yö\n[J]uokse");
84          } else if (uusiPaikka == Paikka.MONSTERITAISTELUTAPPIO || uusiPaikka == Paikka.PAHOLAINENTAPPIO) {
85              komentoValikko.setText("[J]atka peliä\n");
86          } else if (uusiPaikka == Paikka.TAISTELUAREENAEI || uusiPaikka == Paikka.METSAPUPU) {
87              komentoValikko.setText("[T]akaisin");
88          } else if (uusiPaikka == Paikka.AREENATAISTELU || uusiPaikka == Paikka.PAHOLAINEN) {
89              komentoValikko.setText("[L]yö" + lisaaApuKomento(uusiPaikka));
90          } else if (uusiPaikka == Paikka.TAISTELUAREENATULOS || uusiPaikka == Paikka.RUOSTEINENAVAIN) {
91              komentoValikko.setText("[J]atka");
92          } else if (uusiPaikka == Paikka.KAPAKKA || uusiPaikka == Paikka.KANNI) {
93              komentoValikko.setText("[O]sta\n[T]akaisin");
94          } else if (uusiPaikka == Paikka.LUOLA) {
95              komentoValikko.setText("[E]tsi Punainen Paholainen\n[T]akaisin metsään");
96          } else if (uusiPaikka == Paikka.KAPAKKAOSTO) {
97              String ostettavatAvut = new Kapakka(pelaaja).ostokomennot(pelaaja);
98              komentoValikko.setText(ostettavatAvut);
99          } else if (uusiPaikka == Paikka.PARANTAJA) {
100             komentoValikko.setText(lisaaParantajaKomento());
101         } else if (uusiPaikka == Paikka.CASINO) {
102             String komennot = new Casino().getKomennot(pelaaja);
103             komentoValikko.setText(komennot);
104         } else if (uusiPaikka == Paikka.PEUKKUPELI) {
105             komentoValikko.setText("[A]loita\n[T]akaisin");
106         } else if (uusiPaikka == Paikka.PEUKKUTULOS) {
107             komentoValikko.setText("[T]akaisin");
108         } else if (uusiPaikka == Paikka.HEIKENNETTYPAHOLAINEN) {
109             komentoValikko.setText(lisaaHeikennettyPaholainenKomento());
110         } else if (uusiPaikka == Paikka.LOPPU) {
111             komentoValikko.setText("[V]iimeinen lyönti");
112         }
113     }
114 
115     private String lisaaApuKomento(Paikka uusiPaikka) {
116         if (uusiPaikka == Paikka.AREENATAISTELU && pelaaja.onkoPelaajallaApu(new OhraPotion(pelaaja))) {
117             return "\n[O]hraPotion";
118         } else if (uusiPaikka == Paikka.PAHOLAINEN && pelaaja.onkoPelaajallaApu(new RuosteinenAvain())) {
119             return "\n[R]uosteinen avain";
120         } else {
121             return "";
122         }
123     }
124 
125     private String lisaaHeikennettyPaholainenKomento() {
126         String komennot = "[L]yö";
127         if (luola.getPunainenPotion() > 0) {
128             komennot += "\n[P]unainen potion";
129         }
130         if (luola.getMustaPotion() > 0) {
131             komennot += "\n[M]usta potion";
132         }
133         return komennot;
134     }
135 
136     private String lisaaMetsaKomennot() {
137         String metsaKomento = "[E]tsi monsteri";
138         if (pelaaja.getTaso() == 10) {
139             metsaKomento += "\n[P]unainen Paholainen";
140         }
141         metsaKomento += "\n[V]ointiPotion (" + pelaaja.getPotionit() + ")";
142         return metsaKomento + "\n[T]akaisin kylään";
143     }
144 
145     private String lisaaAseenMyyntiKomento() {
146         String komento = "";
147         if (new Asepaja().voikoMyydaAseen(pelaaja)) {
148             komento += "[K]yllä\n[E]i";
149         } else {
150             komento += "[T]akaisin";
151         }
152         return komento;
153     }
154 
155     private String lisaaParantajaKomento() {
156         String komento = "";
157         if (new Parantaja().voikoOstaa(pelaaja)) {
158             komento += "[P]aranna\n[O]sta VointiPotion\n[T]akaisin";
159         } else {
160             komento += "[P]aranna\n[T]akaisin";
161         }
162         return komento;
163     }

167         super.paintComponent(g);
168     }
