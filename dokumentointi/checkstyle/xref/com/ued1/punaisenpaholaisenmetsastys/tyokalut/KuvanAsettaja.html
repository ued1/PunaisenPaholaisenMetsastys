1   package com.ued1.punaisenpaholaisenmetsastys.tyokalut;
2   
3   import java.awt.image.BufferedImage;
4   import java.util.Map;
5   import javax.swing.ImageIcon;
6   import javax.swing.JLabel;
7   
8   /**
9    * Luokan tehtävänä on asettaa kuvia ja ikoneita JLabel-olioihin.
10   */
11  public class KuvanAsettaja {
12  
13      private Map<String, BufferedImage> kuvat;
14      private Map<String, BufferedImage> ikonit;
15  
16      public KuvanAsettaja() {
17          kuvat = new KuvanLataaja().lataaKuvat();
18          ikonit = new KuvanLataaja().lataaIkonit();
19      }
20  
21      /**
22       * Metodi asettaa kuvausta (kuvia talletettavan Map-olion key) vastaavan
23       * kuvan JLabeliin ja palauttaa sen kutsujalle. Kuvaus asetetaan kuvan
24       * ToolTip-tekstiksi. Mikäli kuvaa ei löydy, asettaa metodi kuvauksen
25       * JLabeliin ilman kuvaa.
26       *
27       * @param kuvaus kuvan map-key ja ToolTip-teksti
28       * @return JLabel joko kuvalla tai tekstillä
29       */
30      public JLabel asetaIkoni(String kuvaus) {
31          JLabel label;
32          if (ikonit.containsKey(kuvaus)) {
33              label = new JLabel(new ImageIcon(ikonit.get(kuvaus)));
34              label.setToolTipText(kuvaus);
35          } else {
36              label = new JLabel(kuvaus + ":");
37          }
38          return label;
39      }
40  
41      /**
42       * Metodi asettaa kuvan parametrina saatuun JLabel-olioon mikäli kuva löytyy
43       * Map-oliosta kuvausta vastaavalla key:llä. Mikäli kuva löytyy, asetetaan
44       * kuvaus ToolTip-tekstiksi. Mikäli kuvausta ei löydy, palautetaan JLabel
45       * kuvausta vastaavalla tekstillä ilman kuvaa.
46       *
47       * @param label JLabel-olio, johon kuva halutaan asettaa
48       * @param kuvaus Merkkijono, jonka perusteella kuva etsitään
49       */
50      public void asetaKuva(JLabel label, String kuvaus) {
51          if (kuvat.containsKey(kuvaus)) {
52              label.setIcon(new ImageIcon(kuvat.get(kuvaus)));
53              label.setToolTipText(kuvaus);
54              label.setText(null);
55          } else {
56              label.setIcon(null);
57              label.setToolTipText(null);
58              label.setText(kuvaus);
59          }
60  
61      }
62  
63  }
