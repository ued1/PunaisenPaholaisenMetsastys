19      /**
20       * Lataa pelaajatietopaneelissa ikonit ja palauttaa ne Map-oliona.

22       * @return ladatut ikonit map-oliona
23       */
24      public Map lataaIkonit() {
25          lisaaIkonit();
26          return kuvat;
27      }
28  
29      /**
30       * Lataa pelissa näytettävät kuvat ja palauttaa ne Map-oliona.
31       * 
32       * @return ladatus kuvat map-oliona
33       */
34      public Map lataaKuvat() {
35          lisaaKuvat();
36          return kuvat;
37      }

83      private void lisaa(String sijainti, String kuvaus) {
84          try {
85              BufferedImage kuva = ImageIO.read(this.getClass().getResource(sijainti));
86              kuvat.put(kuvaus, kuva);
87          } catch (Exception e) {
88              // Ei tehdä mitään
89          }
90      }
