1   package com.ued1.punaisenpaholaisenmetsastys.peli.logiikka;
2   
3   import com.ued1.punaisenpaholaisenmetsastys.peli.Vaikeus;
4   import com.ued1.punaisenpaholaisenmetsastys.peli.haarniskat.Riepu;
5   import com.ued1.punaisenpaholaisenmetsastys.peli.hahmot.Hahmo;
6   import com.ued1.punaisenpaholaisenmetsastys.peli.hahmot.Monsteri;
7   import com.ued1.punaisenpaholaisenmetsastys.peli.hahmot.Pelaaja;
8   
9   /**
10   * Luokka Luola määrittelee luolan toiminnallisuuden. Luola on pelin loppu ja
11   * ilmestyy metsään kun pelaaja on korkeimmalla mahdollisella tasolla.
12   */
13  public class Luola {
14  
15      private Pelaaja pelaaja;
16      private Taistelu taistelu;
17      private Hahmo paholainen;
18      private int punainenPotion;
19      private int mustaPotion;
20  
21      public Luola(Pelaaja pelaaja) {
22          this.pelaaja = pelaaja;
23          this.taistelu = null;
24          this.punainenPotion = 0;
25          this.mustaPotion = 0;
26          if (pelaaja.getVaikeus() == Vaikeus.HELPPO) {
27              this.paholainen = new Monsteri(500, 150, 65, "Punainen Paholainen");
28          } else {
29              this.paholainen = new Monsteri(1000, 150, 65, "Punainen Paholainen");
30          }
31      }
32  
33      /**
34       * Metodi alustaa taistelun Punaista Paholaista vastaan.
35       */
36      public void haastaPaholainen() {
37          taistelu = new Taistelu(pelaaja, paholainen);
38          punainenPotion = 0;
39      }
40  
41      /**
42       * Metodi palauttaa meneillään olevan taistelun.
43       *
44       * @return meneillään oleva Taistelu
45       */
46      public Taistelu getTaistelu() {
47          return taistelu;
48      }
49  
50      /**
51       * Metodi asettaa taistelun tuloksen. Pelaajan hävitessä rahat laskevat
52       * nollaan ja pelaaja menettää haarniskan. Pelaaja kuitenkin parantuu ja voi
53       * jatkaa peliä. Pelaajan voittessa peli loppuu.
54       */
55      public void asetaTulos() {
56          if (!pelaaja.onkoElossa()) {
57              pelaaja.setHaarniska(new Riepu());
58              pelaaja.muutaRahoja(-pelaaja.getRahat());
59              pelaaja.paranna();
60              paholainen.paranna();
61              mustaPotion = 0;
62              punainenPotion = 0;
63          }
64      }
65  
66      /**
67       * Asettaa Ruosteisen avaiman avulla löytämien potionien määrän.
68       */
69      public void asetaPotionit() {
70          mustaPotion = 5;
71          punainenPotion = 5;
72      }
73  
74      /**
75       * Pelaaja käyttää Ruosteisen avaimen avulla löytämänsä punaisen potionin.
76       * Punainen potion parantaa pelaajan.
77       */
78      public void kaytaPunainenPotion() {
79          if (punainenPotion > 0) {
80              pelaaja.paranna();
81              punainenPotion--;
82          }
83      }
84  
85      /**
86       * Pelaaja käyttää Ruosteista avaimen avulla löytämänsä mustan potionin.
87       * Musta potion heikentaa pelaajaa.
88       */
89      public void kaytaMustaPotion() {
90          if (mustaPotion > 0) {
91              laskePelaajanVointia(100);
92              laskePelaajanVointia(10);
93              laskePelaajanVointia(1);
94              mustaPotion--;
95          }
96  
97      }
98  
99      private void laskePelaajanVointia(int paljonko) {
100         if (pelaaja.getVointi() - paljonko > 0) {
101             pelaaja.setVointi(pelaaja.getVointi() - paljonko);
102         }
103     }
104 
105     /**
106      * Metodi palauttaa Punaista Paholaista vastaan käytettävien punaisten
107      * potionien määrän.
108      *
109      * @return kokonaisluku, punaisten potionien määrä
110      */
111     public int getPunainenPotion() {
112         return punainenPotion;
113     }
114 
115     /**
116      * Metodi palauttaa Punaista Paholaista vastaan käytettävien mustien
117      * potionien määrän.
118      *
119      * @return kokonaisluku, mustien potionien määrä
120      */
121     public int getMustaPotion() {
122         return mustaPotion;
123     }
124 
125 }
