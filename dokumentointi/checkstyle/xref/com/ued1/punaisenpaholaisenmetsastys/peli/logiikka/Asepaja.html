22  /**
23   * Luokka hoitaa pelin asepajan logiikan.
24   */
25  public class Asepaja extends Kauppa {
26  
27      /**
28       * Konstruktori antaa valmiin hinnaston yläluokalle (Kauppa) kutsuessaan
29       * yläluokan konstruktoria.
30       */
31      public Asepaja() {
32          super(lisaaAseetValikoimaan());
33      }
34  
35      private static ArrayList<Ase> lisaaAseetValikoimaan() {
36          ArrayList<Ase> aseet = new ArrayList<>();
37          aseet.add(new Nyrkki()); // oletusase, ei näy valikoimassa
38          aseet.add(new Keppi());
39          aseet.add(new Maila());
40          aseet.add(new Puukko());
41          aseet.add(new Leka());
42          aseet.add(new Tikari());
43          aseet.add(new Kirves());
44          aseet.add(new Hukari());
45          aseet.add(new Floretti());
46          aseet.add(new Wakizashi());
47          aseet.add(new Gladius());
48          aseet.add(new Sapeli());
49          aseet.add(new Pitkamiekka());
50          aseet.add(new Katana());
51          aseet.add(new Excalibur());
52          return aseet;
53      }
54  
55      /**
56       * Metodia käytetään aseen ostamiseen.
57       *
58       * @param pelaaja Pelaaja, jolle ase ostetaan
59       * @param aseenNumero Asevalikoimassa näkyvän aseen numero
60       * @return oston onnistumisesta kertova totuusarvo, true jos osto onnistui
61       */
62      @Override
63      public boolean osta(Pelaaja pelaaja, int aseenNumero) {
64          if (aseenNumero >= getValikoima().size() || aseenNumero < 0) {
65              return false;
66          } else if (voikoOstaaAseen(pelaaja, (Ase) getValikoima().get(aseenNumero))) {
67              pelaaja.setAse((Ase) getValikoima().get(aseenNumero));
68              pelaaja.muutaRahoja(0 - (((Ase) getValikoima().get(aseenNumero)).arvo()));
69              return true;
70          }
71          return false;
72      }
73  
74      /**
75       * Metodia käytetään aseen myymiseen.
76       *
77       * @param pelaaja Pelaaja, jonka ase halutaan myydä
78       * @return myynnin onnistumisesta kertova totuusarvo, true jos myynti
79       * onnistui
80       */
81      public boolean myy(Pelaaja pelaaja) {
82          if (voikoMyydaAseen(pelaaja)) {
83              pelaaja.muutaRahoja(pelaaja.getAse().arvo() / 2); // 50% takaisin vanhasta aseesta
84              pelaaja.setAse(new Nyrkki());
85              return true;
86          }
87          return false;
88      }
89  
90      /**
91       * Metodi tarkistaa voiko pelaaja myydä aseensa. Myynti on mahdollista kun
92       * pelaajan aseena on mikä tahansa muu ase kuin nyrkki.
93       *
94       * @param pelaaja Pelaaja, jonka asetta ollaan myymässä
95       * @return totuusarvo, true jos myynti on mahdollinen
96       */
97      public boolean voikoMyydaAseen(Pelaaja pelaaja) {
98          if (pelaaja.getAse().toString().equals("Nyrkki")) {
99              return false;
100         }
101         return true;
102     }
103 
104     /**
105      * Metodi tarkistaa voiko pelaaja ostaa tietyn aseen. Osto on mahdollinen
106      * jos pelaajalla on riittävästi rahaa ja hänellä on aseena nyrkki.
107      *
108      * @param pelaaja Pelaaja, jonka ostokyky tarkistetaan
109      * @param ase Ase, jota ollaan ostamassa
110      * @return totuusarvo, true jos osto on mahdollinen
111      */
112     public boolean voikoOstaaAseen(Pelaaja pelaaja, Ase ase) {
113         if (pelaaja.getRahat() >= ase.arvo() && (pelaaja.getAse().toString().equals("Nyrkki"))) {
114             return true;
115         }
116         return false;
117     }
118 
119     /**
120      * Metodi palauttaa asepajan hinnaston otsikoineen. Aseet ovat numeroituina
121      * ja hinnoiteltuina.
122      *
123      * @return asepajan hinnasto merkkijonona
124      */
125     @Override
126     public String valikoimaMerkkijonona() {
127         return String.format("%-19s%-10s%-7s", "Aseen nimi", "Hinta", "Voima") + "\n" + super.valikoimaMerkkijonona();
128     }
