21   * Luokka hoitaa pelin haarniskakaupan logiikan.
22   */
23  public class Haarniskakauppa extends Kauppa {
24  
25      public Haarniskakauppa() {
26          super(lisaaHaarniskatValikoimaan());
27      }
28  
29      private static ArrayList<Haarniska> lisaaHaarniskatValikoimaan() {
30          ArrayList<Haarniska> haarniskat = new ArrayList<>();
31          haarniskat.add(new Riepu());
32          haarniskat.add(new Vaatteet()); // 1
33          haarniskat.add(new Takki()); // 2
34          haarniskat.add(new Nahkatakki());
35          haarniskat.add(new Rengashaarniska());
36          haarniskat.add(new Pronssihaarniska());
37          haarniskat.add(new Rautahaarniska());
38          haarniskat.add(new Titanhaarniska());
39          haarniskat.add(new Timanttihaarniska());
40          haarniskat.add(new BatmanAsu());
41          haarniskat.add(new PotterinViitta()); // 10
42          haarniskat.add(new PomminPurkuPuku()); // 11
43          haarniskat.add(new PinkitTrigoot()); // 12
44          return haarniskat;
45      }
46  
47      /**
48       * Metodi tarkistaa voiko pelaaja ostaa tietyn haarniskan. Osto on
49       * mahdollinen jos pelaajalla on riittävästi rahaa.
50       *
51       * @param pelaaja Pelaaja, joka on haarniskaa ostamassa
52       * @param haarniska Haarniska, jota pelaaja on ostamassa
53       * @return totuusarvo, true jos osto on mahdollinen
54       */
55      public boolean voikoOstaaHaarniskan(Pelaaja pelaaja, Haarniska haarniska) {
56          if (pelaaja.getRahat() < haarniska.arvo()) {
57              return false;
58          }
59          return true;
60      }
61  
62      /**
63       * Haarniskan ostamiseen tarkoitettu metodi.
64       *
65       * @param pelaaja Pelaaja, jolle haarniska ostetaan
66       * @param haarniskanNumero Ostettavan haarniskan numero hinnastosta
67       * @return totuusarvo, true jos osto onnistui
68       */
69      @Override
70      public boolean osta(Pelaaja pelaaja, int haarniskanNumero) {
71          if (voikoPelaajaOstaaOstoksen(pelaaja, haarniskanNumero)) {
72              Haarniska haarniska = (Haarniska) getValikoima().get(haarniskanNumero);
73              pelaaja.setHaarniska(haarniska);
74              pelaaja.muutaRahoja(0 - haarniska.arvo());
75              return true;
76          }
77          return false;
78      }
79  
80      /**
81       * Metodi palauttaa haarniskakaupan hinnaston merkkijonona, haarniskat
82       * numeroituina ja hinnoiteltuina.
83       *
84       * @return hinnasto merkkijonona
85       */
86      @Override
87      public String valikoimaMerkkijonona() {
88          return "Myynnissä oleva haarniska  Puolustusvoima\n" + super.valikoimaMerkkijonona();
89      }
90  
91  }
