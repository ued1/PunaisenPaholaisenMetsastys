3   import com.ued1.punaisenpaholaisenmetsastys.peli.hahmot.Pelaaja;
4   import java.util.Random;
5   
6   /**
7    * Pupu on "Apu", joka varastaa pelaajan.
8    */
9   public class Pupu extends Apu {
10  
11      private Pelaaja pelaaja;
12  
13      public Pupu(Pelaaja pelaaja) {
14          this.pelaaja = pelaaja;
15      }
16  
17      @Override
18      public String toString() {
19          return "Pupu";
20      }
21  
22      /**
23       * Pupu "auttaa" pelaajaa varastamalla pelaajan rahat 1/15
24       * todennäköisyydellä kun pelaaja on aloittamassa monsteritaistelua.
25       * Pupu ei "auta" mikäli pelaajalla ei ole rahaa.
26       *
27       * @return totuusarvo, true jos pupu vie rahat
28       */
29      @Override
30      public boolean auta() {
31          Random arpoja = new Random();
32          if (pelaaja.getRahat() > 0 && arpoja.nextInt(16) == 5) {
33              pelaaja.muutaRahoja(-pelaaja.getRahat());
34              pelaaja.poistaApu(this);
35              return true;
36          }
37          return false;
38      }
39  
40      @Override
41      public String kuvaus() {
42          return "\nPupu varasti kaikki rahasi\nja loikki tiehensä!";
43      }
44  
45      @Override
46      public int arvo() {
47          return 10;
48      }
49  
50  }
