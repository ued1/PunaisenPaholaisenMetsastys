1   package com.ued1.punaisenpaholaisenmetsastys.peli.apuvalineet;

3   /**
4    * Apuvälineet auttavat pelaajaa eri tilanteissa. Joistakin avuista voi olla
5    * haittaa hyödyn sijaan.
6    */
7   public abstract class Apu {
8   
9       /**
10       * Palauttaa avun nimen.
11       *
12       * @return avun nimi merkkijonona
13       */
14      @Override
15      public abstract String toString();
16  
17      /**
18       * Metodi auta auttaa pelaajaa. Jokainen apu määrittelee itse miten pelaajaa
19       * autetaan. Metodi pelauttaa totuusarvon jos auttaminen onnistui.
20       *
21       * @return totuusarvo true jos auttaminen onnistui
22       */
23      public abstract boolean auta();
24  
25      /**
26       * Metodi palauttaa tarinapaneeliin sopivan kuvauksen siitä mitä tapahtuu.
27       *
28       * @return tarinapaneeliin kuvaus merkkijonona
29       */
30      public abstract String kuvaus();
31  
32      /**
33       * Metodi palauttaa avun arvon, mitä tarvitaan kapakassa apuja ostaessa.
34       *
35       * @return avun arvo
36       */
37      public abstract int arvo();
38  
39      @Override
40      public boolean equals(Object olio) {
41          return this.toString().equals(((Apu) olio).toString());
42      }
43  
44      @Override
45      public int hashCode() {
46          int hash = 7;
47          return hash;
48      }
49  
50  }
